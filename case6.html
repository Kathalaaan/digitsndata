    PokemonID INT PRIMARY KEY,

CREATE TABLE Trainers (
    TrainerID INT PRIMARY KEY,
    Name VARCHAR(255),
    Hometown VARCHAR(255),
    PokemonID INT,
    FOREIGN KEY (PokemonID) REFERENCES Pokémon(PokemonID)
);
